  9 FileResult asset_load_file(const char *file_path, const char *access) {
 10   FILE *file = fopen(file_path, access);
 11   FileResult res = {};
 12   if (file == NULL) {
 13     fprintf(stderr, "Failed to open file: %s", file_path);
 14     return res;
 15   }
 16 
 17   char *source = NULL;
 18   uint32_t lenght = 0;
 19   /* Go to the end of the file. */
 20   if (fseek(file, 0L, SEEK_END) == 0) {
 21     /* Get the size of the file. */
 22     long bufsize = ftell(file);
 23     if (bufsize == -1) { /* Error */
 24       return res;
 25     }
 26 
 27     /* Allocate our buffer to that size. */
 28     source = malloc(sizeof(char) * (bufsize + 1));
 29 
 30     /* Go back to the start of the file. */
 31     if (fseek(file, 0L, SEEK_SET) != 0) { /* Error */
 32       exit(1);
 33     }
 34 
 35     /* Read the entire file into memory. */
 36     size_t newLen = fread(source, sizeof(char), bufsize, file);
 37     if (ferror(file) != 0) {
 38       fprintf(stderr, "Error reading file: %s", file_path);
 39     } else {
 40       source[newLen++] = '\0'; /* Just to be safe. */
 41     }
 42     lenght = bufsize;
 43   }
 44   fclose(file);
 45   res.contents = source;
 46   res.lenght = lenght;
 47   return res;
 48 }

 50 static void get_file_data(void *ctx, const char *filename, const int is_mtl,
 51                           const char *obj_filename, char **data, size_t *len) {
 52   // NOTE: If you allocate the buffer with malloc(),
 53   // You can define your own memory management struct and pass it through `ctx`
 54   // to store the pointer and free memories at clean up stage(when you quit an
 55   // app)
 56   // This example uses mmap(), so no free() required.
 57   (void)ctx;
 58 
 59   if (!filename) {
 60     fprintf(stderr, "null filename\n");
 61     (*data) = NULL;
 62     (*len) = 0;
 63     return;
 64   }
 65 
 66   FileResult file = asset_load_file(filename, "rb");
 67   (*data) = file.contents;
 68   (*len) = file.lenght;
 69 }

 71 Mesh *asset_load_mesh(const char *file_path) {
 72   tinyobj_attrib_t attrib;
 73   tinyobj_shape_t *shapes = NULL;
 74   size_t num_shapes;
 75   tinyobj_material_t *materials = NULL;
 76   size_t num_materials;
 77 
 78   unsigned int flags = TINYOBJ_FLAG_TRIANGULATE;
 79   char *content = NULL;
 80   int ret = tinyobj_parse_obj(&attrib, &shapes, &num_shapes, &materials,
 81                               &num_materials, file_path, get_file_data, content,
 82                               flags);
 83   if (ret != TINYOBJ_SUCCESS) {
 84     PANIC("failed to load asset");
 85   }
 86 
 87   Vertex *vertices = malloc(sizeof(Vertex) * attrib.num_vertices);
 88   for (int i = 0; i < attrib.num_vertices; i++) {
 89     Vertex *vertex = &vertices[i];
 90     tinyobj_vertex_index_t *v = &attrib.faces[i];
 91     vertex->pos[0] = attrib.vertices[v->v_idx + 0];
 92     vertex->pos[1] = attrib.vertices[v->v_idx + 1];
 93     vertex->pos[2] = attrib.vertices[v->v_idx + 2];
 94     glm_vec3_copy((vec3){0.0f, 0.0f, 0.0f}, vertex->color);
 95   }
 96 
 97   for (int i = 0; i < attrib.num_normals; i++) {
 98     Vertex *vertex = &vertices[i];
 99     tinyobj_vertex_index_t *v = &attrib.faces[i];
100 
101     vertex->normal[0] = attrib.normals[v->vn_idx + 0];
102     vertex->normal[1] = attrib.normals[v->vn_idx + 1];
103     vertex->normal[2] = attrib.normals[v->vn_idx + 2];
104   }
105 
106   /*Mesh *mesh = mesh_create_prealloced(
107       attrib.num_vertices / *attrib.face_num_verts, attrib.num_vertices);*/
108 
109   Mesh *mesh = mesh_create_prealloced(vertices, attrib.num_vertices, NULL, 0);
110   tinyobj_attrib_free(&attrib);
111   tinyobj_materials_free(materials, num_materials);
112   tinyobj_shapes_free(shapes, num_shapes);
113 
114   if (content != NULL) {
115     free(content);
116   }
117 
118   return mesh;
119 }

121 Texture asset_load_texture(const TextureCreateInfo *info,
122                            const char *file_path) {
123   Texture texture = texture_create(info);
124   int width, height, nr_channels;
125   unsigned char *data = stbi_load(file_path, &width, &height, &nr_channels, 0);
126   if (data) {
127     glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, width, height, 0, GL_RGB,
128                  GL_UNSIGNED_BYTE, data);
129     glGenerateMipmap(GL_TEXTURE_2D);
130   } else {
131     fprintf(stderr, "failed to load image: %s\n", file_path);
132   }
133   stbi_image_free(data);
134 
135   return texture;
136 }
